def initialize(reason = nil)
  if reason.nil? || reason.empty?
    super('Malformed Container.')
  elsif reason.is_a?(Array)
    reasons = reason.map { |r| " * #{r}\n" }
    super("Malformed Container:\n#{reasons}")
  else
    super("Malformed Container: #{reason}")
  end
end
